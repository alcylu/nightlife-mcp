const supabase = await createSupabaseServerClient();
const { data: { user: caller }, error: authError } = await supabase.auth.getUser();
if (authError || !caller) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

const { data: callerRoles } = await supabase.from('user_roles').select('role').eq('user_id', caller.id);
const isAdmin = callerRoles?.some(r => r.role === 'super_admin' || r.role === 'admin');
if (!isAdmin) return NextResponse.json({ error: 'Forbidden: admin role required' }, { status: 403 });
