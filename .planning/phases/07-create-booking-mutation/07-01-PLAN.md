```typescript
import { createVipAdminBooking } from '@/services/vipAdminService';
import { z } from 'zod';

const createVipAdminBookingSchema = z.object({
  venue_id: z.string().uuid(),
  booking_date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
  arrival_time: z.string().regex(/^([01]\d|2[0-3]):([0-5]\d)$/),
  party_size: z.coerce.number().int().min(1).max(30),
  customer_name: z.string().min(1),
  customer_email: z.string().email(),
  customer_phone: z.string().min(1),
  preferred_table_code: z.string().optional(),
  special_requests: z.string().optional(),
  agent_internal_note: z.string().optional(),
  change_note: z.string().optional(),
});

export async function POST(request: Request) {
  try {
    // 1. Auth (same as GET)
    const supabase = await createSupabaseServerClient();
    const { data: { user: caller }, error: authError } = await supabase.auth.getUser();
    if (authError || !caller) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }

    // 2. Role check (same as GET)
    const { data: callerRoles } = await supabase
      .from('user_roles').select('role').eq('user_id', caller.id);
    const isAdmin = callerRoles?.some(r => r.role === 'super_admin' || r.role === 'admin');
    if (!isAdmin) {
      return NextResponse.json({ error: 'Forbidden: admin role required' }, { status: 403 });
    }

    // 3. Parse + validate body
    const body = await request.json();
    const parsed = createVipAdminBookingSchema.safeParse(body);
    if (!parsed.success) {
      return NextResponse.json(
        { error: parsed.error.issues.map(i => i.message).join('; ') },
        { status: 400 },
      );
    }

    // 4. Derive editor_username
    const editorUsername = caller.email ?? caller.id.slice(0, 8);

    // 5. Call service
    const serviceClient = createServiceRoleClient();
    const result = await createVipAdminBooking(serviceClient, {
      ...parsed.data,
      editor_username: editorUsername,
    });

    return NextResponse.json(result, { status: 201 });
  } catch (error: unknown) {
    const message = error instanceof Error ? error.message : 'Internal server error';
    console.error('Unexpected error in POST /api/vip/bookings:', error);
    return NextResponse.json({ error: message }, { status: 500 });
  }
}
```

Add the hook after the existing `useVipVenues()`:

```typescript
export function useCreateVipBooking() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (input: CreateVipAdminBookingInput): Promise<VipBookingCreateResult> => {
      const response = await fetch('/api/vip/bookings', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(input),
      });
      if (!response.ok) {
        const err = await response.json().catch(() => ({ error: response.statusText }));
        throw new Error(err.error ?? `HTTP ${response.status}`);
      }
      return response.json();
    },
    onSuccess: (data) => {
      queryClient.invalidateQueries({ queryKey: ['vip-bookings'] });
      const tableWarnSuffix = data.table_warning ? ` (${data.table_warning})` : '';
      toast.success(`Booking created successfully${tableWarnSuffix}`);
    },
    onError: (error: Error) => {
      toast.error(`Failed to create booking: ${error.message}`);
    },
  });
}
```
