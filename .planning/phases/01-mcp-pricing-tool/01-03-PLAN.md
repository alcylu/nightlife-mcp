   ```typescript
   // GET /api/v1/venues/:id/vip-pricing
   router.get("/venues/:id/vip-pricing", async (req, res) => {
     try {
       const result = await getVipPricing(supabase, {
         venue_id: req.params.id,
         date: str(req.query.date),
       });
       res.json(result);
     } catch (error) {
       sendError(res, error);
     }
   });
   ```
