       function matchQuery(
         row: EventOccurrenceRow,
         query: string,
         performers: string[],
         genres: string[],
       ): boolean {
         const needle = query;  // already normalizeQuery output — no further processing
         if (!needle) return true;

         const norm = (s: string | null | undefined) =>
           stripAccents(String(s || "")).toLowerCase().replace(/\s+/g, "");

         const venue = firstRelation(row.venue);
         if (
           [
             row.name_en, maybeJa(row.name_i18n),
             row.description_en, maybeJa(row.description_i18n),
             venue?.name, venue?.name_en, venue?.name_ja,
             venue?.city, venue?.city_en, venue?.city_ja,
           ].some((v) => norm(v).includes(needle))
         ) {
           return true;
         }

         return (
           performers.some((name) => norm(name).includes(needle)) ||
           genres.some((name) => norm(name).includes(needle))
         );
       }
